001package org.xbib.elasticsearch.plugin.reference;
002
003import org.elasticsearch.common.inject.Module;
004import org.elasticsearch.plugins.AbstractPlugin;
005import org.xbib.elasticsearch.module.reference.ReferenceModule;
006
007import java.util.Collection;
008
009import static org.elasticsearch.common.collect.Lists.newArrayList;
010
011public class ReferencePlugin extends AbstractPlugin {
012
013    @Override
014    public String name() {
015        return "analysis-reference-"
016                + Build.getInstance().getVersion() + "-"
017                + Build.getInstance().getShortHash();
018    }
019
020    @Override
021    public String description() {
022        return "Analysis reference plugin";
023    }
024
025    @Override
026    public Collection<Class<? extends Module>> indexModules() {
027        Collection<Class<? extends Module>> modules = newArrayList();
028        modules.add(ReferenceModule.class);
029        return modules;
030    }
031
032}
