035public class ReferenceMappingTest extends Assert {
036
037    private DocumentMapperParser mapperParser;
038
039    private Node node;
040
041    private Client client;
042
043    @BeforeClass
044    public void setupMapperParser() throws IOException {
045        Settings nodeSettings = ImmutableSettings.settingsBuilder()
046                .put("gateway.type", "none")
047                .put("index.store.type", "memory")
048                .put("index.number_of_shards", 1)
049                .put("index.number_of_replica", 0)
050                .put("cluster.routing.schedule", "50ms")
051                .build();
052        node = NodeBuilder.nodeBuilder().settings(nodeSettings).local(true).build().start();
053        client = node.client();
054
055        BytesReference json = jsonBuilder().startObject().array("myfield", "a","b","c").endObject().bytes();
056        client.prepareIndex("test", "test", "1234").setSource(json).execute().actionGet();
057
058        json = jsonBuilder().startObject().field("author", "John Doe").endObject().bytes();
059        client.prepareIndex("test", "authorities", "1").setSource(json).execute().actionGet();
060
061        Index index = new Index("test");
062        Map<String, AnalyzerProviderFactory> analyzerFactoryFactories = Maps.newHashMap();
063        analyzerFactoryFactories.put("keyword",
064                new PreBuiltAnalyzerProviderFactory("keyword", AnalyzerScope.INDEX, new KeywordAnalyzer()));
065        Settings settings = ImmutableSettings.Builder.EMPTY_SETTINGS;
066        AnalysisService analysisService = new AnalysisService(index, settings, null, analyzerFactoryFactories, null, null, null);
067        mapperParser = new DocumentMapperParser(index, settings,
068                analysisService,
069                new PostingsFormatService(index),
070                new DocValuesFormatService(index),
071                new SimilarityLookupService(index, settings),
072                null);
073        Settings indexSettings = ImmutableSettings.settingsBuilder()
074                .put("ref_index", "test")
075                .put("ref_type", "test")
076                .put("ref_fields", "myfield").build();
077        mapperParser.putTypeParser(ReferenceMapper.CONTENT_TYPE,
078                new ReferenceMapper.TypeParser(client, indexSettings));
079    }
080
081    @AfterClass
082    public void shutdown() {
083        client.close();
084        node.close();
085    }
086
087    @Test
088    public void testRefMappings() throws Exception {
089        String mapping = copyToStringFromClasspath("/ref-mapping.json");
090        DocumentMapper docMapper = mapperParser.parse(mapping);
091        BytesReference json = jsonBuilder().startObject().field("_id", 1).field("someField", "1234").endObject().bytes();
092        ParseContext.Document doc = docMapper.parse(json).rootDoc();
093        assertNotNull(doc);
094        assertNotNull(docMapper.mappers().smartName("someField"));
095        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), "1234");
096        assertEquals(doc.getFields("someField.ref").length, 3);
097        assertEquals(doc.getFields("someField.ref")[0].stringValue(), "a");
098        assertEquals(doc.getFields("someField.ref")[1].stringValue(), "b");
099        assertEquals(doc.getFields("someField.ref")[2].stringValue(), "c");
100
101        // re-parse it
102        String builtMapping = docMapper.mappingSource().string();
103        docMapper = mapperParser.parse(builtMapping);
104
105        json = jsonBuilder().startObject().field("_id", 1).field("someField", "1234").endObject().bytes();
106        doc = docMapper.parse(json).rootDoc();
107
108        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), "1234");
109        assertEquals(doc.getFields("someField.ref").length, 3);
110        assertEquals(doc.getFields("someField.ref")[0].stringValue(), "a");
111        assertEquals(doc.getFields("someField.ref")[1].stringValue(), "b");
112        assertEquals(doc.getFields("someField.ref")[2].stringValue(), "c");
113    }
114
115    @Test
116    public void testRef() throws Exception {
117        String mapping = copyToStringFromClasspath("/ref-mapping-authorities.json");
118        DocumentMapper docMapper = mapperParser.parse(mapping);
119        BytesReference json = jsonBuilder().startObject()
120                .field("_id", 1)
121                .field("title", "A title")
122                .startObject("author")
123                .field("index", "test")
124                .field("type", "authorities")
125                .field("id", "1")
126                .field("fields", "author")
127                .endObject()
128                .endObject().bytes();
129        ParseContext.Document doc = docMapper.parse(json).rootDoc();
130        assertEquals(doc.getFields("author.ref").length, 1);
131        assertEquals(doc.getFields("author.ref")[0].stringValue(), "John Doe");
132    }
